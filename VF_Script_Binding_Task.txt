		set &QuinetteAccuracyScore8Sec expression ( ( &Hits8Sec - &FalseAlarms8Sec ) / &ResponsesGiven8Sec ) * 100
		set &QuinnetteProcessingScore8Sec expression ( ( &Hits8Sec - &NonResponses8Sec ) / 20 ) * 100
		set &Mean_RT_Experiment expression ( &Sum_RT_Experiment / 40 )
		save &AccuracyScoreExperiment &AccuracyPercentExperiment &QuinetteAccuracyScore1Sec &QuinnetteProcessingScore1Sec &QuinetteAccuracyScore8Sec &QuinnetteProcessingScore8Sec &Mean_RT_Experiment
	fi

	set &Sum_RT_Experiment expression &Sum_RT_Experiment + &binding_task_mismatch_8s_RT
	#cálculo da variável que indica se o participante deu uma resposta certa ou errada em cada ensaio (&binding_task_mismatch_8s_accuracy),
	#do número total de acertos (&AccuracyScoreExperiment), dos acertos nos ensaios com um intervalo de retenção de 8 segundos (&Hits8Sec),
	#do número total de respostas dadas nos ensaios com um intervalo de retenção de 8 segundos(&ResponsesGiven8Sec), e do número de ensaios
	#em que o participante não respondeu nos ensaios com um intervalo de retenção de 8 segundos (&NonResponses8Sec)
	if STATUS == CORRECT
		set &binding_task_mismatch_8s_accuracy 1
		set &AccuracyScoreExperiment expression &AccuracyScoreExperiment + 1
		set &Hits8Sec expression &Hits8Sec + 1
		set &ResponsesGiven8Sec expression &ResponsesGiven8Sec + 1
	fi
	if STATUS == WRONG
		set &binding_task_mismatch_8s_accuracy 0
		set &AccuracyScoreExperiment expression &AccuracyScoreExperiment + 0
		set &FalseAlarms8Sec expression &FalseAlarms8Sec + 1
		set &ResponsesGiven8Sec expression &ResponsesGiven8Sec + 1
	fi
	if STATUS == TIMEOUT
		set &binding_task_mismatch_8s_accuracy 2
		set &AccuracyScoreExperiment expression &AccuracyScoreExperiment + 0
		set &NonResponses8Sec expression &NonResponses8Sec + 1
	fi
	save &Ref &binding_task_mismatch_8s_stimuliNumber &binding_task_mismatch_8s_RT &binding_task_mismatch_8s_accuracy
	#calulo do número total de acertos nos ensaios experimentais, da percentagem de acertos no ensaios experimentais
	#e da taxa de acertos e processing score para as condições com um intervalo de retenção de 1 e 8 segundos, segundo os autores originais  
    #da tarefa (Quinette et al. 2006)
	if &CounterExperiment == 40
		set &AccuracyPercentExperiment expression ( &AccuracyScoreExperiment / 40 ) * 100
		set &QuinetteAccuracyScore1Sec expression ( ( &Hits1Sec - &FalseAlarms1Sec ) / &ResponsesGiven1Sec ) * 100
		set &QuinnetteProcessingScore1Sec expression ( ( &Hits1Sec - &NonResponses1Sec ) / 20 ) * 100
		set &QuinetteAccuracyScore8Sec expression ( ( &Hits8Sec - &FalseAlarms8Sec ) / &ResponsesGiven8Sec ) * 100
		set &QuinnetteProcessingScore8Sec expression ( ( &Hits8Sec - &NonResponses8Sec ) / 20 ) * 100
		set &Mean_RT_Experiment expression ( &Sum_RT_Experiment / 40 )
		save &AccuracyScoreExperiment &AccuracyPercentExperiment  &QuinetteAccuracyScore1Sec &QuinnetteProcessingScore1Sec &QuinetteAccuracyScore8Sec &QuinnetteProcessingScore8Sec &Mean_RT_Experiment
	fi
